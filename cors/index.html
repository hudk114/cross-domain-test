1. 简单请求，不需要options 
            
          fetch('http://localhost:31311/cors/get', {
            credentials: 'include'
          }).then(function(response) {
            console.log(response);
          });

          fetch('http://localhost:31311/cors/post', {
            method: 'POST',
            credentials: 'include',
            header: {
              'content-type': 'application/x-www-form-urlencoded'
            },
            body: "name=hudk"
          }).then(function(response) {
            console.log(response);
          });
  2. 非简单请求，需要options 
            
          fetch('http://localhost:31311/cors/get', {
            credentials: 'inclue',
            headers: {
              'Content-Type': 'text/json'
            }
          }).then(function(response) {
            console.log(response);
          });
